import asyncio
import os
import logging
import sys
from pathlib import Path

# =============================================================================
#  سكريبت موحّد لخط الأنابيب الكامل + اختبار الترجمة:
#  فيديو ← استخراج الصوت ← VAD ← STT ← تصدير SRT ← ترجمة ← تحليل دلالي
# =============================================================================

# إعداد الـ Logging المركزي
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    handlers=[logging.StreamHandler(sys.stdout)]
)
logger = logging.getLogger("FullPipeline")

# ---------- استيراد خدمات استخراج الصوت ----------
from audio_extraction.audio_extractor import AudioExtractionService
from audio_extraction.ffmpeg_client import AsyncFFmpegClient

# ---------- استيراد خدمات الـ VAD ----------
from vad.vad_config import SileroVADConfig
from vad.vad_client import SileroVADClient
from vad.vad_service import SileroVADService
from vad.exceptions import VADBaseError, VADInputValidationError

# ---------- استيراد خدمات الـ STT (مع استخدام الحزمة الصغرى "stt") ----------
from STT.stt_client import STTClient
from STT.stt_service import STTService

# ---------- استيراد خدمة تصدير SRT ----------
from exporters import SRTExporter

# ---------- استيراد خدمة الترجمة ----------
from translation.translation_service import TranslationService
from translation.exceptions import TranslationError

# ---------- استيراد خدمة التحليل الدلالي بالذكاء الاصطناعي ----------
from analytics import AIAnalyzerService, AIAnalysisBaseError


async def run_full_pipeline():
    logger.info("🚀 بدء تشغيل خط الأنابيب الكامل (فيديو ← SRT ← ترجمة ← تحليل ذكي)...")

    # =====================================================================
    # تحقق استباقي من المتغيرات البيئية (Fail Fast)
    # =====================================================================
    api_key = os.getenv("GITHUB_TOKEN")
    if not api_key:
        logger.error("❌ متغير البيئة GITHUB_TOKEN غير موجود!")
        print("\n[تنبيه أمني/تشغيلي]: يرجى تعيين GITHUB_TOKEN قبل التشغيل.")
        print("مثال: os.environ['GITHUB_TOKEN'] = 'your_token_here' أو عبر الملف البيئي .env\n")
        return

    # =====================================================================
    # المرحلة الأولى: استخراج الصوت من الفيديو
    # =====================================================================
    logger.info("▶️ المرحلة 1: استخراج الصوت من الفيديو")

    video_file = "5666799983_fb.mp4"  # <-- ضع مسار الفيديو الخاص بك هنا
    output_directory = "./output_audio"

    ffmpeg_client = AsyncFFmpegClient()
    extraction_service = AudioExtractionService(ffmpeg_client=ffmpeg_client)

    try:
        audio_path = await extraction_service.process_video(video_file, output_directory)
        logger.info(f"✅ تم استخراج الصوت بنجاح: {audio_path}")
    except Exception as e:
        logger.error(f"❌ فشل استخراج الصوت: {e}")
        return

    # =====================================================================
    # المرحلة الثانية: Voice Activity Detection (VAD)
    # =====================================================================
    logger.info("▶️ المرحلة 2: تحليل الصوت واستخراج فترات الكلام (VAD)")

    vad_config = SileroVADConfig()
    vad_client = SileroVADClient(config=vad_config)
    vad_service = SileroVADService(config=vad_config, client=vad_client)

    logger.info("⏳ جاري تحميل نموذج Silero VAD...")
    await vad_client.load_model()
    logger.info("✅ تم تحميل النموذج بنجاح.")

    try:
        speech_segments = await vad_service.process_audio_file(Path(audio_path))
        logger.info(f"✅ تم اكتشاف {len(speech_segments)} مقطع صوتي يحتوي على كلام.")
    except (VADBaseError, VADInputValidationError) as e:
        logger.error(f"❌ خطأ في مرحلة VAD: {e}")
        return
    except Exception as e:
        logger.error(f"❌ خطأ غير متوقع في VAD: {e}")
        return

    if not speech_segments:
        logger.warning("⚠️ لم يتم اكتشاف أي كلام في الملف الصوتي. إنهاء خط الأنابيب.")
        return

    # =====================================================================
    # المرحلة الثالثة: Speech-to-Text (STT)
    # =====================================================================
    logger.info("▶️ المرحلة 3: تحويل الكلام إلى نص (STT)")

    stt_client = STTClient()
    stt_service = STTService(stt_client=stt_client)

    language = "en"  # جرّب تغييرها إلى "ar" أو أي لغة أخرى

    try:
        logger.info("⏳ جاري تشغيل STT على المقاطع المستخرجة...")
        final_transcript = await stt_service.process_audio(
            audio_path=audio_path,
            speech_segments=speech_segments,
            language=language
        )
        logger.info("✅ اكتمل تحويل النص بنجاح.")
    except Exception as e:
        logger.error(f"❌ خطأ في مرحلة STT: {e}")
        return

    # =====================================================================
    # المرحلة الرابعة: تصدير ملف الترجمة SRT
    # =====================================================================
    logger.info("💾 المرحلة 4: إنشاء ملف SRT...")

    if not final_transcript:
        logger.error("❌ لم يتم استخراج أي نص. لا يمكن إنشاء ملف SRT.")
        return

    exporter = SRTExporter()
    output_srt = "outputs/subtitles/final_subtitle.srt"

    try:
        saved_path = exporter.export(data=final_transcript, output_path=output_srt)
        logger.info(f"✅ تم حفظ ملف SRT في: {saved_path}")

        # قراءة محتوى SRT للمرحلة التالية
        with open(saved_path, 'r', encoding='utf-8') as f:
            srt_content = f.read()
    except Exception as e:
        logger.error(f"❌ فشل تصدير ملف SRT: {e}")
        return

    # =====================================================================
    # المرحلة الخامسة: ترجمة ملف SRT
    # =====================================================================
    logger.info("🌐 المرحلة 5: ترجمة ملف SRT (English → Arabic)...")

    translation_service = TranslationService(
        api_base_url="https://models.inference.ai.azure.com",
        api_key=api_key,
        model="gpt-4o",
        chunk_size=50
    )

    translated_srt_path = None
    try:
        translated_srt = await translation_service.translate_srt(
            srt_content=srt_content,
            source_lang="English",
            target_lang="Arabic"
        )

        # حفظ ملف SRT المترجم
        translated_srt_path = "outputs/subtitles/final_subtitle_arabic.srt"
        os.makedirs(os.path.dirname(translated_srt_path), exist_ok=True)
        with open(translated_srt_path, 'w', encoding='utf-8') as f:
            f.write(translated_srt)

        logger.info(f"✅ تمت الترجمة وحفظ الملف المترجم في: {translated_srt_path}")
    except TranslationError as e:
        logger.error(f"❌ فشل الترجمة: {e}")
        # نستمر في التحليل الدلالي حتى لو فشلت الترجمة
    except Exception as e:
        logger.error(f"❌ خطأ غير متوقع في الترجمة: {e}")
    finally:
        await translation_service.close()
        logger.info("🔌 تم إغلاق خدمة الترجمة.")

    # =====================================================================
    # المرحلة السادسة: التحليل الدلالي بالذكاء الاصطناعي
    # =====================================================================
    logger.info("🧠 المرحلة 6: بدء التحليل الدلالي (Semantic Analysis)...")

    # تجميع النص الكامل من المقاطع المستخرجة
    full_transcript_text = " ".join([seg["text"] for seg in final_transcript])

    analyzer = AIAnalyzerService()
    summary_result = None

    try:
        summary_result = await analyzer.generate_summary(full_transcript_text)
        logger.info("✅ تم إنشاء الملخص الدلالي بنجاح.")
    except AIAnalysisBaseError as e:
        logger.error(f"❌ فشل الموديول في معالجة الطلب منطقياً: {e}")
    except Exception as e:
        logger.critical(f"❌ خطأ غير متوقع أثناء التحليل الدلالي: {e}", exc_info=True)
    finally:
        logger.info("🔌 إغلاق عميل الـ HTTP وجاري تحرير الموارد...")
        await analyzer.close()
        logger.info("✅ تم تحرير الموارد بنجاح.")

    # =====================================================================
    # عرض الناتج النهائي ورسالة النجاح
    # =====================================================================
    print("\n" + "=" * 60)
    print("🎉 اكتمل خط الأنابيب بالكامل!")
    print(f"✅ تم حفظ ملف الترجمة الأصلي في:")
    print(f"   {os.path.abspath(saved_path)}")

    if translated_srt_path:
        print(f"✅ تم حفظ ملف الترجمة العربي في:")
        print(f"   {os.path.abspath(translated_srt_path)}")

    print("💡 يمكنك فتح ملفات SRT بأي محرر نصوص أو استخدامها مع مشغل فيديو (مثل VLC).")
    print("=" * 60)

    # عرض مختصر للنص المستخرج
    if final_transcript:
        print("\n🎙️ النص المستخرج مع التوقيت (أول 10 مقاطع):")
        for i, segment in enumerate(final_transcript[:10], 1):
            start = segment['start']
            end = segment['end']
            text = segment['text']
            print(f"[{start:06.2f}s -> {end:06.2f}s] | {text}")
        if len(final_transcript) > 10:
            print(f"... و {len(final_transcript) - 10} مقاطع إضافية.")

    # عرض نتائج التحليل الدلالي (إن وجدت)
    if summary_result:
        print("\n" + "-" * 60)
        print("💡 [الملخص القصير - Short Summary]:")
        print(f"   {summary_result.short_summary}")
        print("-" * 60)
        print("📝 [النقاط التفصيلية - Detailed Summary (Exactly 4 Points)]:")
        for idx, point in enumerate(summary_result.detailed_summary, 1):
            print(f"   {idx}. {point}")
        print("-" * 60 + "\n")


if __name__ == "__main__":
    asyncio.run(run_full_pipeline())
